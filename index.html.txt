    const districtSelect = document.getElementById("districtSelect");
    const subjectSelect = document.getElementById("subjectSelect");
    const zoneSelect = document.getElementById("zoneSelect");
    const manualButton = document.getElementById("manualButton");
    const searchInput = document.getElementById("searchInput");
    const searchResults = document.getElementById("searchResults");

    const resultCard = document.getElementById("resultCard");
    const resultTitle = document.getElementById("resultTitle");
    const resultDistrict = document.getElementById("resultDistrict");
    const resultSubject = document.getElementById("resultSubject");
    const resultZone = document.getElementById("resultZone");

    function normalize(value) {
      return String(value || "")
        .toLowerCase()
        .replaceAll("ё", "е")
        .trim();
    }

    function zonesFor(subjectName) {
      return ZONES[subjectName] || DEFAULT_ZONES;
    }

    function districtForSubject(subjectName) {
      return DISTRICTS.find((district) => district.subjects.includes(subjectName));
    }

    function fillDistricts() {
      DISTRICTS.forEach((district, index) => {
        const option = document.createElement("option");
        option.value = String(index);
        option.textContent = district.name;
        districtSelect.appendChild(option);
      });
    }

    function fillSubjects() {
      subjectSelect.innerHTML = "";
      zoneSelect.innerHTML = "";
      manualButton.disabled = true;

      if (districtSelect.value === "") {
        subjectSelect.disabled = true;
        zoneSelect.disabled = true;
        subjectSelect.innerHTML = '<option value="">Сначала выберите округ</option>';
        zoneSelect.innerHTML = '<option value="">Сначала выберите субъект</option>';
        return;
      }

      const district = DISTRICTS[Number(districtSelect.value)];

      subjectSelect.disabled = false;
      subjectSelect.innerHTML = '<option value="">Выберите субъект РФ</option>';

      district.subjects.forEach((subject) => {
        const option = document.createElement("option");
        option.value = subject;
        option.textContent = subject;
        subjectSelect.appendChild(option);
      });

      zoneSelect.disabled = true;
      zoneSelect.innerHTML = '<option value="">Сначала выберите субъект</option>';
    }

    function fillZones() {
      zoneSelect.innerHTML = "";
      manualButton.disabled = true;

      if (!subjectSelect.value) {
        zoneSelect.disabled = true;
        zoneSelect.innerHTML = '<option value="">Сначала выберите субъект</option>';
        return;
      }

      zoneSelect.disabled = false;
      zoneSelect.innerHTML = '<option value="">Выберите зону / подрегион</option>';

      zonesFor(subjectSelect.value).forEach((zone) => {
        const option = document.createElement("option");
        option.value = zone;
        option.textContent = zone;
        zoneSelect.appendChild(option);
      });
    }

    function buildSearchIndex() {
      const items = [];

      PLACE_HINTS.forEach((place) => {
        const district = districtForSubject(place.subject);
        if (!district) return;

        items.push({
          type: "place",
          label: place.name,
          district: district.name,
          subject: place.subject,
          zone: place.zone,
          searchText: `${place.name} ${place.subject} ${place.zone} ${district.name}`
        });
      });

      DISTRICTS.forEach((district) => {
        district.subjects.forEach((subject) => {
          zonesFor(subject).forEach((zone) => {
            items.push({
              type: "zone",
              label: `${subject} — ${zone}`,
              district: district.name,
              subject,
              zone,
              searchText: `${subject} ${zone} ${district.name}`
            });
          });
        });
      });

      return items;
    }

    const SEARCH_INDEX = buildSearchIndex();

    function showResult(selection) {
      const title = selection.place || selection.subject;

      resultTitle.textContent = title;
      resultDistrict.textContent = selection.district;
      resultSubject.textContent = selection.subject;
      resultZone.textContent = selection.zone;

      resultCard.hidden = false;
      resultCard.scrollIntoView({ behavior: "smooth", block: "start" });
    }

    function renderSearchResults() {
      const query = normalize(searchInput.value);
      searchResults.innerHTML = "";

      if (query.length < 2) {
        return;
      }

      const matches = SEARCH_INDEX
        .filter((item) => normalize(item.searchText).includes(query))
        .slice(0, 12);

      if (!matches.length) {
        searchResults.innerHTML = `
          <div class="empty-result">
            Ничего не найдено. Выберите федеральный округ, субъект и зону вручную.
          </div>
        `;
        return;
      }

      matches.forEach((item) => {
        const button = document.createElement("button");
        button.type = "button";
        button.className = "result";

        const main = item.type === "place" ? item.label : item.subject;
        const detail = item.type === "place"
          ? `${item.subject} · ${item.zone}`
          : `${item.district} · ${item.zone}`;

        button.innerHTML = `
          <strong>${main}</strong>
          <span>${detail}</span>
        `;

        button.addEventListener("click", () => {
          showResult({
            place: item.type === "place" ? item.label : "",
            district: item.district,
            subject: item.subject,
            zone: item.zone
          });

          searchInput.value = item.type === "place" ? item.label : item.subject;
          searchResults.innerHTML = "";
        });

        searchResults.appendChild(button);
      });
    }

    districtSelect.addEventListener("change", fillSubjects);
    subjectSelect.addEventListener("change", fillZones);

    zoneSelect.addEventListener("change", () => {
      manualButton.disabled = !zoneSelect.value;
    });

    manualButton.addEventListener("click", () => {
      if (!districtSelect.value || !subjectSelect.value || !zoneSelect.value) return;

      const district = DISTRICTS[Number(districtSelect.value)];

      showResult({
        place: "",
        district: district.name,
        subject: subjectSelect.value,
        zone: zoneSelect.value
      });
    });

    searchInput.addEventListener("input", renderSearchResults);

    fillDistricts();
  </script>
